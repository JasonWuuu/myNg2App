import {Pipe} from 'angular2/core';


@Pipe({
  name: 'VehicleTypePipe'
})
export class AgePipe {

  transform(input, args?) {
    let [minAge, maxAge] = args;
    return input.filter(person => {
      return person.age >= +minAge && person.age <= +maxAge;
    });
  }
